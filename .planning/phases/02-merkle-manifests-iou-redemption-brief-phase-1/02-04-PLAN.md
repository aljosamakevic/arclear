struct Iou { address debtor; address creditor; uint256 amount; uint256 nonce; uint64 expiry; bytes32 ref; }
// Field order/types byte-match IOU_TYPES in src/domain.ts (expiry IS uint64 — verified).

struct StoredRoot { bytes32 root; uint64 nonce; uint64 executedAt; }

mapping(uint256 => StoredRoot) public rootRing;   // key: nonce % RING
mapping(address => uint64) public lastRound;      // 1-based (nonce+1); 0 = never participated
mapping(bytes32 => bool) public redeemed;         // nullifier, key = IOU id = hashIou(iou) (D-13)

uint64 public immutable K;                 // staleness gate, default 3  — UNCALIBRATED
uint64 public immutable RING;              // root-ring size, default 16 — UNCALIBRATED
uint64 public immutable MAX_IOU_LIFETIME;  // "L" seconds, default 86400 — UNCALIBRATED (D-15 coverage rule)

constructor(IERC20 token_, uint64 k_, uint64 ring_, uint64 maxIouLifetime_)
    EIP712("ArcClearingHub", "1") Ownable(msg.sender)   // domain byte-identical — digest fixtures depend on it

function hashIou(Iou calldata iou) public view returns (bytes32);
function executeRound(uint64 nonce_, address[] calldata participants, int256[] calldata deltas,
                      bytes32[] calldata consumedIds, bytes[] calldata signatures) external;
function redeemIOU(Iou calldata iou, bytes calldata sig,
                   ManifestMerkle.NonInclusionProof[] calldata proofs) external;

bytes32 private constant IOU_TYPEHASH = keccak256(
    "IOU(address debtor,address creditor,uint256 amount,uint256 nonce,uint64 expiry,bytes32 ref)"
);

event IouRedeemed(bytes32 indexed id, address indexed debtor, address indexed creditor,
                  uint256 amount, uint64 atRoundNonce);
