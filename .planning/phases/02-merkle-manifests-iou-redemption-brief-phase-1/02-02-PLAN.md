```solidity
library ManifestMerkle {
    bytes32 internal constant EMPTY_ROOT = keccak256(""); // == viem keccak256("0x"), the v1 sentinel (D-04)

    struct InclusionProof {
        bytes32 leaf;        // raw IOU id (pre-leaf-hash)
        uint256 index;       // 0-based position in the sorted leaf list
        uint256 leafCount;   // total leaves of that round's manifest
        bytes32[] siblings;  // bottom-up; promotion levels consume no sibling
    }

    enum NonInclusionKind { BelowFirst, AboveLast, Bracket } // 0,1,2 — matches TS "belowFirst"/"aboveLast"/"bracket"

    struct NonInclusionProof {
        NonInclusionKind kind;
        InclusionProof a;    // BelowFirst: first leaf | AboveLast: last leaf | Bracket: lower neighbor
        InclusionProof b;    // Bracket only: upper neighbor (ignored otherwise)
    }

    error UnsortedLeaves(uint256 index); // ids[index] <= ids[index-1]

    function rootOf(bytes32[] memory ids) internal pure returns (bytes32);
    function verifyInclusion(InclusionProof memory p, bytes32 root) internal pure returns (bool);
    function verifyNonInclusion(bytes32 id, NonInclusionProof memory p, bytes32 root) internal pure returns (bool);
}
```

Hashing: leaf = `keccak256(abi.encodePacked(bytes1(0x00), id))`; node = `keccak256(abi.encodePacked(bytes1(0x01), left, right))` — ordered, never sorted. Verification walk and non-inclusion rules exactly as specified in RESEARCH Pattern 3 / the 02-01 interfaces block: strict inequalities, all-siblings-consumed check, `index < leafCount`, sentinel short-circuit (`root == EMPTY_ROOT` ⇒ non-inclusion holds).
